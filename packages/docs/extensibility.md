# Extensibility & Customization

Kompo is designed to be extensible at multiple levels while maintaining a clean separation between your business logic and the tooling code.

## Architecture Overview

Kompo enforces a strict separation of concerns:

```
┌─────────────────────────────────────┐
│         Business Code               │  ← Your product logic
│    (apps, domains, shared)          │  ← Generated by Kompo
│         Immutable                   │  ← Never put tooling here
├─────────────────────────────────────┤
│      Extension Zone                 │  ← Your customization space
│         (.kompo/)                   │  ← Override templates, plugins
│         Fully Customizable          │
├─────────────────────────────────────┤
│        Kompo Core                   │  ← Internal engine
│        (packages/)                  │  ← Do not modify
│         Read-only                   │
└─────────────────────────────────────┘
```

## Extension Zones

### 1. Local Templates (`.kompo/templates/`)

Override any template used by the CLI to match your project's standards.

**Example: Add license headers to all generated files**

Create `.kompo/templates/add/domain/index.eta`:
```html
/**
 * © <%= it.org || 'My Company' %> 2024 - All Rights Reserved
 * Domain: <%= it.pascalName %>
 * Generated by Kompo CLI
 */

export * from './entities'
export * from './value-objects'
export * from './ports'
export * from './use-cases'

// Custom domain exports
export type { <%= it.pascalName %>Config } from './config'
```

**Template Priority Order:**
1. `.kompo/templates/` (your overrides)
2. `enterprise/cli/src/templates/` (enterprise templates)
3. `packages/cli/src/templates/` (community templates)

### 2. Local Plugins (`.kompo/plugins/`)

Create project-specific generators and commands without publishing to npm.

**Structure:**
```
.kompo/plugins/
└── my-custom-generator/
    ├── index.ts           # Plugin entry point
    ├── package.json       # Plugin metadata
    └── templates/         # Plugin templates
        └── custom.eta
```

**Plugin Example:**
```typescript
// .kompo/plugins/my-custom-generator/index.ts
import { Plugin } from '@kompo/cli-kit'

export default class MyCustomGenerator implements Plugin {
  name = 'my-custom-generator'
  description = 'Generates custom API adapter'
  
  async activate() {
    // Register commands
    this.registerCommand('generate-api', this.generateApi.bind(this))
  }
  
  private async generateApi(options: any) {
    // Custom generation logic
  }
}
```

**Register in `kompo.config.json`:**
```json
{
  "plugins": [
    "./.kompo/plugins/my-custom-generator"
  ]
}
```

### 3. NPM Plugins

Install community or enterprise plugins from npm.

```bash
# Community plugins
pnpm add -D @kompo/plugin-stripe
pnpm add -D @kompo/plugin-prisma

# Enterprise plugins
pnpm add -D @kompo/enterprise-auth
```

## Customization Examples

### Custom Domain Template

Create a standardized domain structure for your team:

`.kompo/templates/add/domain/entities/entity.eta`:
```html
/**
 * <%= it.pascalName %> Entity
 * @module <%= it.domainName %>
 */

import { <%= it.pascalName %>Id } from '../value-objects/<%= it.pascalName %>Id'

export type <%= it.pascalName %> = {
  id: <%= it.pascalName %>Id
  createdAt: Date
  updatedAt: Date
<% it.fields.forEach(field => { %>
  <%= field.name %>: <%= field.type %><% if (field.optional) { %> | null<% } %>
<% }) %>
}

export function create<%= it.pascalName %>(input: {
<% it.fields.forEach(field => { %>
  <%= field.name %>: <%= field.type %><% if (field.optional) { %>?<% } %>
<% }) %>): <%= it.pascalName %> {
  return {
    id: generate<%= it.pascalName %>Id(),
    createdAt: new Date(),
    updatedAt: new Date(),
    ...input
  }
}
```

### Custom Adapter Generator

Create a generator for your internal APIs:

`.kompo/plugins/internal-api/index.ts`:
```typescript
import { Plugin, Generator } from '@kompo/cli-kit'

export default class InternalApiPlugin implements Plugin {
  name = 'internal-api'
  description = 'Generates adapters for internal APIs'
  
  async activate() {
    this.registerGenerator('internal-api', new InternalApiGenerator())
  }
}

class InternalApiGenerator extends Generator {
  async generate(options: { name: string; version: string }) {
    const { name, version } = options
    
    // Generate adapter
    await this.render('adapters/internal-api.eta', {
      name,
      version,
      pascalName: toPascalCase(name)
    })
    
    // Generate types
    await this.render('types/api-types.eta', {
      name,
      version
    })
  }
}
```

### Custom CLI Commands

Add project-specific commands:

`.kompo/plugins/dev-tools/index.ts`:
```typescript
import { Plugin } from '@kompo/cli-kit'

export default class DevToolsPlugin implements Plugin {
  name = 'dev-tools'
  description = 'Development utilities'
  
  async activate() {
    this.registerCommand('db:reset', this.resetDatabase.bind(this))
    this.registerCommand('dev:setup', this.setupDev.bind(this))
  }
  
  private async resetDatabase() {
    // Custom database reset logic
    await this.exec('pnpm db:drop && pnpm db:create && pnpm db:migrate')
  }
  
  private async setupDev() {
    // Setup development environment
    await this.exec('pnpm install && pnpm db:setup && ppm dev')
  }
}
```

### kompo.config.json

```json
{
  "version": 1,
  "project": {
    "name": "my-app",
    "org": "@mycompany"
  },
  "plugins": [
    "./.kompo/plugins/dev-tools",
    "./.kompo/plugins/internal-api",
    "@kompo/plugin-stripe"
  ],
  "templates": {
    "licenseHeader": "© My Company 2024",
    "author": "Development Team",
    "includeTests": true,
    "testFramework": "vitest"
  },
  "standards": {
    "naming": {

## Best Practices

### 1. Keep Business Logic Separate

Never put business logic in `.kompo/`. This directory is for tooling only.

```typescript
// ❌ Wrong - Business logic in .kompo
.kompo/plugins/payment-plugin/payment-logic.ts

// ✅ Correct - Tooling only
.kompo/plugins/payment-plugin/generator.ts
```

### 2. Version Your Custom Templates

```
.kompo/templates/
├── v1/
│   └── add/
│       └── domain/
└── v2/
    └── add/
        └── domain/
```

### 3. Document Customizations

Create `.kompo/README.md`:
```markdown
# Project Customizations

## Templates
- Domain structure includes validation helpers
- All entities include audit fields

## Plugins
- `dev-tools`: Database reset and dev setup
- `internal-api`: Generates API adapters

## Usage
```bash
pnpm kompo db:reset
pnpm kompo generate-api --name user-service --version v2
```
```

### 4. Test Custom Templates

```typescript
// .kompo/tests/template.test.ts
import { renderTemplate } from '@kompo/cli-kit'

test('custom domain template', async () => {
  const result = await renderTemplate('add/domain/index.eta', {
    domainName: 'user',
    pascalName: 'User'
  })
  
  expect(result).toContain('© My Company')
  expect(result).toContain('export type { UserConfig }')
})
```

## Advanced Extensions

### Custom Validators

```typescript
// .kompo/plugins/validators/index.ts
export default class ValidatorPlugin implements Plugin {
  async activate() {
    this.registerValidator('custom-email', (value: string) => {
      return value.endsWith('@mycompany.com')
    })
  }
}
```

### Code Transformers

```typescript
// .kompo/plugins/transformers/index.ts
export default class TransformerPlugin implements Plugin {
  async activate() {
    this.registerTransformer('add-logging', (code: string) => {
      return wrapWithLogging(code)
    })
  }
}
```

### Integration Hooks

```typescript
// .kompo/plugins/hooks/index.ts
export default class HooksPlugin implements Plugin {
  async activate() {
    this.hooks.before('add:domain', async (options) => {
      await this.ensureDirectoryExists('types')
    })
    
    this.hooks.after('add:entity', async (entity) => {
      await this.generateCrudOperations(entity)
    })
  }
}
```

## Sharing Customizations

### As NPM Package

```bash
# Create package
mkdir my-kompo-extensions
cd my-kompo-extensions
npm init -y

# Add plugins
mkdir -p plugins/{theme,validators}
```

```json
// package.json
{
  "name": "@mycompany/kompo-extensions",
  "version": "1.0.0",
  "kompo": {
    "plugins": [
      "./plugins/theme",
      "./plugins/validators"
    ]
  }
}
```

### As Git Repository

```bash
# Clone into .kompo/plugins
git clone https://github.com/mycompany/kompo-extensions.git \
  .kompo/plugins/external-extensions
```
